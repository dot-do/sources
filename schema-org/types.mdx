```typescript
// Example: "Organization" → "organization"
function generateTypeSlug(label: string): string {
  return label
    .replace(/([A-Z])/g, '-$1') // CamelCase → kebab-case
    .toLowerCase()
    .replace(/^-/, '') // Remove leading hyphen
}

// Examples:
// "Thing" → "thing"
// "Organization" → "organization"
// "LocalBusiness" → "local-business"
// "SoftwareApplication" → "software-application"
```

## Transform Logic

```typescript
interface SchemaTypeFile {
  $id: string
  $type: string
  label: string
  comment: string
  'rdfs:label': string
  'rdfs:subClassOf': string
}

interface Type {
  schemaId: string
  label: string
  description: string
  parentType: string
  parentUrl: string
  slug: string
  properties: string[]
  subTypes: string[]
  collection: 'Types'
  source: 'schema.org'
}

function transformType(file: SchemaTypeFile): Type {
  return {
    schemaId: file.$id,
    label: file.label,
    description: file.comment.trim(),
    parentType: extractTypeName(file['rdfs:subClassOf']),
    parentUrl: file['rdfs:subClassOf'],
    slug: generateTypeSlug(file.label),
    properties: [], // Extract from content
    subTypes: [], // Lookup children
    collection: 'Types',
    source: 'schema.org',
  }
}

function extractTypeName(url: string): string {
  return url.split('/').pop() || 'Thing'
}
```
